## Settings modal

Allows users to configure prayer calculation method and location.

### Props

```typescript
interface SettingsModalProps {
  hidden: boolean;                    // Controls visibility
  setHidden: React.Dispatch<boolean>; // Toggle function
}
```

### Implementation

<CodeGroup>
```tsx components/settings.tsx
import { Settings } from "@/lib/settings";
import {
  ModalOverlay,
  ModalCard,
  ModalHeader,
  ModalBody,
  ModalClose,
} from "./modal";

export default function SettingsModal({
  hidden,
  setHidden,
}: {
  hidden: boolean;
  setHidden: React.Dispatch<boolean>;
}) {
  if (hidden) return <></>;

  return (
    <>
      <ModalOverlay />
      <ModalCard>
        <ModalHeader>
          <h2>Settings</h2>
          <ModalClose onClick={() => setHidden(true)} />
        </ModalHeader>
        <ModalBody>
          {/* Method selector */}
          <select
            onChange={(e) => {
              Settings.method = e.target.value;
              window.location.reload();
            }}
            defaultValue={Settings.method}
          >
            {Object.entries(Settings.methods).map(([key, name]) => (
              <option key={key} value={key}>
                {name}
              </option>
            ))}
          </select>

          {/* Offset selector */}
          <select
            onChange={(e) => {
              Settings.offset = parseInt(e.target.value);
              window.location.reload();
            }}
            defaultValue={Settings.offset}
          >
            {Object.entries(Settings.offsets).map(([key, name]) => (
              <option key={key} value={key}>
                {name}
              </option>
            ))}
          </select>
        </ModalBody>
      </ModalCard>
    </>
  );
}
```
</CodeGroup>

### Usage

```tsx
const [settingsHidden, setSettingsHidden] = useState(true);

<SettingsModal
  hidden={settingsHidden}
  setHidden={setSettingsHidden}
/>
```

<Note>
Changing settings triggers a full page reload to ensure all time calculations are updated with the new method/location.
</Note>

## Duas container

Tabbed interface for displaying Ramadan duas.

### Props

```typescript
// No props - uses hardcoded duas array
export function DuasContainer(): JSX.Element
```

### Data structure

```tsx components/duas.tsx
const Duas = [
  {
    title: "Dua Sahar",
    arabic: "وَبِصَوْمِ غَدٍ نَّوَيْتُ مِنْ شَهْرِ رَمَضَانَ",
    english: "I intend to keep the fast for tomorrow in the month of Ramadan",
    roman: "Wa bisawmi ghadinn nawaiytu min shahri ramadan",
  },
  // ... more duas
];
```

### Implementation

<CodeGroup>
```tsx components/duas.tsx
import classNames from "classnames";

export function DuasContainer() {
  const [activeTab, setActiveTab] = useState(-1);

  return (
    <div className="w-full">
      {/* Tab buttons */}
      <div className="flex gap-2">
        {Duas.map((dua, idx) => (
          <button
            onClick={() => setActiveTab(idx !== activeTab ? idx : -1)}
            className={classNames(
              "flex-1 rounded-xl py-2.5 text-center",
              {
                "bg-[#e8e2d8] text-ink-secondary": activeTab !== idx,
                "bg-gradient-to-r from-[#0f4a2c] to-[#1a7a4e] text-white":
                  activeTab === idx,
              }
            )}
          >
            {dua.title}
          </button>
        ))}
      </div>

      {/* Dua content */}
      {activeTab >= 0 && (
        <div className="card mt-3 p-5">
          <p dir="rtl" className="font-arabic text-ink">
            {Duas[activeTab].arabic}
          </p>
          <div className="divider" />
          <p className="italic text-ink-secondary">
            {Duas[activeTab].roman}
          </p>
          <div className="divider" />
          <p className="text-ink">
            {Duas[activeTab].english}
          </p>
        </div>
      )}
    </div>
  );
}
```
</CodeGroup>

<Note>
Clicking an active tab again collapses it (`idx !== activeTab ? idx : -1`).
</Note>

## Progress bar

Simple progress indicator for fasting progress.

### Props

```typescript
interface ProgressBarProps {
  value: number;  // Progress percentage (0-100)
}
```

### Implementation

```tsx components/progress_bar.tsx
export default function ProgressBar({ value }: { value: number }) {
  return (
    <div className="w-full">
      <div className="h-1.5 w-full overflow-hidden rounded-full bg-border-light">
        <div
          className="h-full rounded-full bg-accent transition-all duration-700 ease-out"
          style={{ width: `${Math.min(Math.max(value, 0), 100)}%` }}
        />
      </div>
      <p className="mt-2 text-center text-xs tabular-nums text-ink-muted">
        {value.toFixed(1)}%
      </p>
    </div>
  );
}
```

### Usage

```tsx
<ProgressBar value={45.7} />
// Displays 45.7% filled bar
```

<Accordion title="Why Math.min and Math.max?">
Ensures value is clamped between 0-100 to prevent visual bugs:
```tsx
Math.min(Math.max(value, 0), 100)
// value = -5  → 0
// value = 50  → 50
// value = 150 → 100
```
</Accordion>

## Print button

Toggles between color and black & white print modes.

### Props

```typescript
// No props - controls print directly
export default function PrintButton(): JSX.Element
```

### Implementation

```tsx components/printButtons.tsx
"use client";

export default function PrintButton() {
  return (
    <div className="hide-print flex gap-x-2">
      <button
        onClick={() => {
          document.body.classList.remove("print-bw");
          window.print();
        }}
      >
        Print Color
      </button>
      <button
        onClick={() => {
          document.body.classList.add("print-bw");
          window.print();
        }}
      >
        Print B&W
      </button>
    </div>
  );
}
```

<Note>
The `hide-print` class hides these buttons in print view using CSS `@media print`.
</Note>
